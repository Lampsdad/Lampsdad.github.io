import React, {useState, useEffect} from 'react';
import { MilkFoamList } from '../helpers/MenuList';
import { MilkTeaList } from '../helpers/MenuList';
import { SlushList } from '../helpers/MenuList';
import { CoffeeList } from '../helpers/MenuList';
import { TeaList } from '../helpers/MenuList';
import { CreativeList } from '../helpers/MenuList';
import MenuItem from '../components/CashierItem';


import '../styles/Cashier.css';

/**
 * Cashier component responsible for displaying various menu categories.
 * @returns {JSX.Element} Cashier component JSX
 */
function Cashier() {
  const [toggle, setToggle] = useState(1);

  const [SeasonalList, setSeasonalList] = useState([]);

    useEffect(() => {
        async function sendRequest(){
            try{
                const body = {"query": "select * from menu_item where id > 54;"}
                const response = await fetch("/query", {
                  method: "POST",
                  headers: {"Content-Type": "application/json"},
                  body: JSON.stringify(body)
                });
                const jsonData = await response.json();
                setSeasonalList(jsonData);
              }
              catch (err){
                console.error(err.message);
              }
        } 
        sendRequest();
    }, []);

   /**
   * Updates the current category to be displayed.
   * @param {number} id - Category ID
   */
  function updateToggle(id) {
    setToggle(id);
  }

  return (
        <div className="menu">
            <h1 className="menuTitle">Order Menu</h1>
            <div className="container">
                <ul className="cc-flex">
                <li className="ctab" onClick={()=>updateToggle(1)}>Milk Foam</li>
                <li className="ctab" onClick={()=>updateToggle(2)}>Milk Tea</li>
                <li className="ctab" onClick={()=>updateToggle(3)}>Slush</li>
                <li className="ctab" onClick={()=>updateToggle(4)}>Coffee</li>
                <li className="ctab" onClick={()=>updateToggle(5)}>Tea</li>
                <li className="ctab" onClick={()=>updateToggle(6)}>Creative</li>
                <li className="ctab" onClick={()=>updateToggle(7)}>Seasonal</li>
                </ul>
                <div className={toggle === 1 ? "menuList" : "hide-menuList"}>
                    {MilkFoamList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
                <div className={toggle === 2 ? "menuList" : "hide-menuList"}>
                    {MilkTeaList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
                <div className={toggle === 3 ? "menuList" : "hide-menuList"}>
                    {SlushList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
                <div className={toggle === 4 ? "menuList" : "hide-menuList"}>
                    {CoffeeList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
                <div className={toggle === 5 ? "menuList" : "hide-menuList"}>
                    {TeaList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
                <div className={toggle === 6 ? "menuList" : "hide-menuList"}>
                    {CreativeList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
                <div className={toggle === 7 ? "menuList" : "hide-menuList"}>
                    {SeasonalList.map((menuItem, key) => {
                        return <MenuItem key={key} name={menuItem.name} price={menuItem.price} id={menuItem.id} />;
                    })}
                </div>
            </div>
        </div>
  )
}

export default Cashier
