import React, {useState} from 'react';
import ManagerInventoryReport from '../components/ManagerInventoryReport';
import ManagerSalesReport from '../components/ManagerSalesReport';
import ManagerExcessReport from '../components/ManagerExcessReport';
import ManagerRestockReport from '../components/ManagerRestockReport';
import ManagerPairReport from '../components/ManagerPairReport';
import '../styles/Manager.css';
import ManagerAddMenuItem from '../components/ManagerAddMenuItem';

/**
 * Manager component to display various reports and functionalities.
 * @returns {JSX.Element} Manager component JSX
 */
function Manager() 
{
  const [toggle, setToggle] = useState(1);

  /**
   * Updates the toggle state based on the selected tab.
   * @param {number} id - ID of the tab to switch to
   */
  function updateToggle(id) {
    setToggle(id);
}

return (
  <div className="reports">
      <h1 className="reportsTitle">Manager Reports</h1>
      <div className="container">
          <ul className="c-flex">
            <li className="tab" onClick={()=>updateToggle(1)}>Inventory</li>
            <li className="tab" onClick={()=>updateToggle(2)}>Sales Report</li>
            <li className="tab" onClick={()=>updateToggle(3)}>Excess Report</li>
            <li className="tab" onClick={()=>updateToggle(4)}>Restock Report</li>
            <li className="tab" onClick={()=>updateToggle(5)}>Sales Pair Report</li>
            <li className="tab" onClick={()=>updateToggle(6)}>Seasonal Items</li>
          </ul>
          <div className={toggle === 1 ? "reportsList" : "hide-reportsList"}>     
            <ManagerInventoryReport></ManagerInventoryReport>
          </div>
          <div className={toggle === 2 ? "reportsList" : "hide-reportsList"}>
              <ManagerSalesReport></ManagerSalesReport>
          </div>
          <div className={toggle === 3 ? "reportsList" : "hide-reportsList"}>
              <ManagerExcessReport></ManagerExcessReport>
          </div>
          <div className={toggle === 4 ? "reportsList" : "hide-reportsList"}>
              <ManagerRestockReport></ManagerRestockReport>
          </div>
          <div className={toggle === 5 ? "reportsList" : "hide-reportsList"}>
             <ManagerPairReport></ManagerPairReport>
          </div>
          <div className={toggle === 6 ? "reportsList" : "hide-reportsList"}>
             <ManagerAddMenuItem></ManagerAddMenuItem>
          </div>
      </div>
  </div>
)
}
export default Manager
