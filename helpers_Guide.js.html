import React, { useState } from 'react';
import '../styles/Guide.css';

/**
 * Component to translate text content and title based on user-selected language.
 * @returns {JSX.Element} Translation component UI.
 */
const Translate = () => {
  const [text, setText] = useState("You can translate the content of this page by selecting a language in the select box.");
  const [title, setTitle] = useState("My Web Page");
  const [targetLanguage, setTargetLanguage] = useState("EN"); // Default to French

  /**
   * Handles translation of text content and title based on the selected language.
   * Fetches translations from Google Translate API and updates state accordingly.
   * @returns {void}
   */
  const handleTranslate = () => {
    const url = "https://translation.googleapis.com/language/translate/v2";
    const apiKey = process.env.REACT_APP_TRANSLATE_TOKEN;

    const params = new URLSearchParams({
      q: [text, title].join('\n'), // Separate text with a newline character
      target: targetLanguage,
      key: apiKey,
    });

    fetch(`${url}?${params}`)
      .then(response => response.json())
      .then(data => {
        setText(data.data.translations[0].translatedText);
        setTitle(data.data.translations[1].translatedText);
      })
      .catch(error => console.error('Translation error:', error));
  };


  return (
    <div className="guideBox">
      <p id="textField">{text}</p>
      <h1 id="title">{title}</h1>
      <p>Hello everybody!</p>
      <p>Translate this page:</p>
      <form>
        <select
          id="targetLanguage"
          value={targetLanguage}
          onChange={e => setTargetLanguage(e.target.value)}
        >
          <option value="ZH">Chinese (Mandarin)</option>
          <option value="CS">Czech</option>
          <option value="DA">Danish</option>
          <option value="NL">Dutch</option>
          <option value="EN">English</option>
          <option value="ET">Estonian</option>
          <option value="FR">French</option>
        </select>

        <input
          type="button"
          id="translateButton"
          value="Translate"
          onClick={handleTranslate}
        />
      </form>
    </div>
  );
};

export default Translate;
