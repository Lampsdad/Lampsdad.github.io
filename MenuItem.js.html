import React from 'react';
import { Link } from 'react-router-dom';


/**
 * Functional component representing a menu item.
 * Displays the image, name, and price of the menu item in a Link component.
 *
 * @param {string} image - The URL of the menu item's image.
 * @param {string} name - The name of the menu item.
 * @param {number} price - The price of the menu item.
 * @param {number} id - The unique identifier of the menu item.
 * @returns {JSX.Element} JSX for the MenuItem component.
 */
function MenuItem({ image, name, price, id}) {
  return (
    <Link className='menuItem' to={{pathname:"/customization"}} state = {{name, price, id}}>
        <div style={{ backgroundImage: `url(${image})` }}> </div>
        <h1> {name} </h1>
        <p> ${price} </p>
    </Link>
  )
}

export default MenuItem
