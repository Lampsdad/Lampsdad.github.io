import { Navigate } from "react-router-dom";
import { useAuth } from "../helpers/AuthContext";
/**
 * ProtectedRoute component to handle protected routes based on user authentication status.
 * @param {Object} children - The components or routes to be rendered within the protected route.
 * @returns {JSX.Element} JSX for rendering children if the user is authenticated; otherwise, redirects to the home page.
 */
const ProtectedRoute = ({ children }) => {
    const {user} = useAuth();
    
  if (!user) {
    return <Navigate to="/" replace />;
  }
  return children;
};
export default ProtectedRoute;
