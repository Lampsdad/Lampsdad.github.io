import React, {useEffect, useState} from 'react';
import {useGoogleLogin } from '@react-oauth/google'
import axios from 'axios';
import { useAuth } from '../helpers/AuthContext';

/**
 * Functional component for handling user login and displaying user information.
 * @returns {JSX.Element} JSX for the Login component.
 */
function Login() {
    // define the user, login, and logout variables from the useAuth hook
    const { user, login, logout } = useAuth();

    const [profileInfo, setProfileInfo] = useState([]);

    const loginClicked = useGoogleLogin({
      onSuccess: (response) => {
        login(response);
      },
      onError: (error) => {
        console.log(`Login Failed: ${error}`);
      }
    });
  const logoutClicked = () => {
        logout();
    };
  
    useEffect(() => {
      if (user) {
        axios
          .get(`https://www.googleapis.com/oauth2/v1/userinfo?access_token=${user.access_token}`, {
            headers: {
              Authorization: `Bearer ${user.access_token}`,
              Accept: 'application/json'
            }
          })
          .then((response) => {
            console.log(response.data);
            // Update user profile information if needed
            setProfileInfo(response.data);
          })
          .catch((error) => console.log(error));
      }
    }, [user]);

    // This is the return statement for the Login component that will be rendered
    return (
      <div className="oauthInfo">
        {user ? (
          <div className="nameInfo">
            <p>{profileInfo.name}</p>
            <button onClick={logoutClicked}>Sign Out</button>
          </div>
        ) : (
          <button onClick={loginClicked}>Sign in</button>
        )}
      </div>
    );
  }
  
  export default Login
