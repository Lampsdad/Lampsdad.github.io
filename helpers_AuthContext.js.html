import React, { createContext, useContext, useState } from 'react';

/**
 * Context to manage user authentication state and actions.
 * @type {React.Context}
 */
const AuthContext = createContext();

/**
 * Provider component to manage user authentication state and actions.
 * @param {Object} props - React component props.
 * @returns {JSX.Element} JSX for the authentication provider component.
 */
export const AuthProvider = ({ children }) => {
  /**
   * State variable to hold user information.
   * @type {Object | null}
   */
    const [user, setUser] = useState(null);

    /**
   * Function to set user information upon login.
   * @param {Object} userInfo - User information object.
   * @returns {void}
   */
    const login = (userInfo) => {
      setUser(userInfo);
    };
  
    /**
   * Function to clear user information upon logout.
   * @returns {void}
   */
    const logout = () => {
        console.log('user has signed out');
      setUser(null);
    };
    
    /**
   * Provide the authentication context and its value to child components.
   * @returns {JSX.Element} JSX for the authentication context provider.
   */
    return (
      <AuthContext.Provider value={{ user, login, logout }}>
        {children}
      </AuthContext.Provider>
    );
  };

  /**
 * Hook to access authentication context values.
 * @returns {Object} Authentication context values (user, login, logout).
 */
  export const useAuth = () => {
    return useContext(AuthContext);
  };
