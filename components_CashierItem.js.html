import React from 'react';
import { Link } from 'react-router-dom';

/**
 * Functional component representing an item in the cashier.
 * @param {Object} props - The properties passed to the CashierItem component.
 * @param {string} props.image - The image URL for the item.
 * @param {string} props.name - The name of the item.
 * @param {number} props.price - The price of the item.
 * @param {number} props.id - The ID of the item.
 * @returns {JSX.Element} JSX for the CashierItem component.
 */
function CashierItem({ image, name, price, id}) {
  return (
    <Link className='cashierItem' to={{pathname:"/customization"}} state = {{name, price, id}}>
        <h1> {name} </h1>
        <p> ${price} </p>
    </Link>
  )
}

export default CashierItem
