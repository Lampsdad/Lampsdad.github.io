import { useEffect, useState } from "react";


/**
 * Functional component for generating a restock report of inventory items with low quantity.
 * @returns {JSX.Element} JSX for the ManagerRestockReport component.
 */
function ManagerRestockReport(){

    const [restockData, setRestockData] = useState([]);

    useEffect(() => {
        async function sendRequest(){
            try{
                const body = {"query": "select * from inventory_item where quantity < 2000 order by id;"}
                const response = await fetch("/query", {
                  method: "POST",
                  headers: {"Content-Type": "application/json"},
                  body: JSON.stringify(body)
                });
                const jsonData = await response.json();
                setRestockData(jsonData);
              }
              catch (err){
                console.error(err.message);
              }
        } 
        sendRequest();
    }, []);

return (
    <table>
        <thead>
        <tr>
            <th>ID</th>
            <th>Name</th>
            <th>Quantity</th>
        </tr>
        </thead>
        <tbody>
        {restockData.map(data => (
            <tr>
            <td>{data.id + " | "}</td>
            <td>{data.name + " | "}</td>
            <td>{data.quantity}</td>
            </tr>
        ))}
        </tbody>
    </table>
)

}  

export default ManagerRestockReport;
